// ===== AUTH (ระบบจัดการเอง) ===== function switchTab(tab) { document.querySelectorAll('.auth-tab').forEach((t,i) => t.classList.toggle('active', (i===0&&tab==='login')||(i===1&&tab==='register'))); document.getElementById('login-form').style.display = tab==='login' ? 'block' : 'none'; document.getElementById('register-form').style.display = tab==='register' ? 'block' : 'none'; clearAlert('alert-auth'); } function showAlert(id, msg, type='error') { const el = document.getElementById(id); el.className = `alert alert-${type} show`; el.textContent = msg; } function clearAlert(id) { const el = document.getElementById(id); el.className = 'alert'; el.textContent = ''; } // เช็คสถานะการล็อกอินเดิม window.onload = async () => { const savedUserId = localStorage.getItem('payub_user_id'); if (savedUserId) { const { data } = await sb.from('profiles').select('*').eq('id', savedUserId).single(); if (data) { currentUser = { id: data.id }; currentProfile = data; initApp(); } } }; async function login() { const email = document.getElementById('login-email').value.trim(); const pass = document.getElementById('login-password').value; if (!email || !pass) return showAlert('alert-auth', 'กรุณากรอกข้อมูลให้ครบ'); // ค้นหาอีเมลและรหัสผ่านในตาราง profiles const { data, error } = await sb.from('profiles').select('*').eq('email', email).eq('password', pass).single(); if (error || !data) return showAlert('alert-auth', 'อีเมลหรือรหัสผ่านไม่ถูกต้อง'); currentUser = { id: data.id }; currentProfile = data; localStorage.setItem('payub_user_id', data.id); // บันทึกการล็อกอินลงเบราว์เซอร์ initApp(); } async function register() { const name = document.getElementById('reg-name').value.trim(); const email = document.getElementById('reg-email').value.trim(); const pass = document.getElementById('reg-password').value; const role = document.getElementById('reg-role').value; if (!name || !email || !pass) return showAlert('alert-auth', 'กรุณากรอกข้อมูลให้ครบ'); if (pass.length < 6) return showAlert('alert-auth', 'รหัสผ่านต้องมีอย่างน้อย 6 ตัวอักษร'); // เช็คว่ามีอีเมลนี้ในระบบหรือยัง const { data: existing } = await sb.from('profiles').select('id').eq('email', email).maybeSingle(); if (existing) return showAlert('alert-auth', 'อีเมลนี้ถูกใช้งานแล้ว'); // บันทึกข้อมูลลงตาราง (เห็นรหัสผ่านได้) const { error } = await sb.from('profiles').insert([{ full_name: name, email: email, password: pass, role: role }]); if (error) return showAlert('alert-auth', error.message); showAlert('alert-auth', 'สมัครสมาชิกสำเร็จ! กรุณาเข้าสู่ระบบ', 'success'); switchTab('login'); } async function logout() { localStorage.removeItem('payub_user_id'); // ลบการจำล็อกอิน document.getElementById('auth-screen').style.display = 'flex'; document.getElementById('app-screen').style.display = 'none'; currentUser = null; currentProfile = null; document.getElementById('login-email').value = ''; document.getElementById('login-password').value = ''; }
