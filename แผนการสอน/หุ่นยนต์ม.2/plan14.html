5. สาระสำคัญ

DC Motor หมุนต่อเนื่อง (ไม่หยุดที่มุม) เหมาะกับล้อหุ่นยนต์ Maker Pi RP2040 มี Motor Driver ชิป DRV8833 ในตัว ขับได้ 2 มอเตอร์ — Motor 1: GP8 (M1A) + GP9 (M1B), Motor 2: GP10 (M2A) + GP11 (M2B) ใช้ PWM frequency 10000 Hz ส่งสัญญาณ duty_cycle (0-65535) เป็นแรง การหมุนทิศ A>0,B=0 = เดินหน้า / A=0,B>0 = ถอยหลัง / A=0,B=0 = หยุด การหมุนตัว (Tank Turn) ใช้ล้อ 2 ฝั่งคนละทิศ การเขียนฟังก์ชันช่วยให้โค้ดอ่านง่ายและใช้ซ้ำได้

เฉลย (สำหรับครู) - โค้ดหลัก

import board, pwmio, time m1a = pwmio.PWMOut(board.GP8, frequency=10000) m1b = pwmio.PWMOut(board.GP9, frequency=10000) m2a = pwmio.PWMOut(board.GP10, frequency=10000) m2b = pwmio.PWMOut(board.GP11, frequency=10000) SPEED = 0.7 def set_motor(a, b, s): if s >= 0: a.duty_cycle = int(s*65535); b.duty_cycle = 0 else: a.duty_cycle = 0; b.duty_cycle = int(abs(s)*65535) def forward(): set_motor(m1a,m1b,SPEED); set_motor(m2a,m2b,SPEED) def backward(): set_motor(m1a,m1b,-SPEED); set_motor(m2a,m2b,-SPEED) def left(): set_motor(m1a,m1b,SPEED); set_motor(m2a,m2b,-SPEED) def right(): set_motor(m1a,m1b,-SPEED); set_motor(m2a,m2b,SPEED) def stop(): set_motor(m1a,m1b,0); set_motor(m2a,m2b,0) forward(); time.sleep(1); backward(); time.sleep(1) left(); time.sleep(0.5); right(); time.sleep(0.5) # 360° rotation - test for actual time left(); time.sleep(2); stop()
