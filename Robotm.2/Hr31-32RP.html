📚 Setup + Motor Functions

import board, pwmio, adafruit_hcsr04, time # --- Setup มอเตอร์ 2 ตัว --- m1a = pwmio.PWMOut(board.GP8, frequency=10000) m1b = pwmio.PWMOut(board.GP9, frequency=10000) m2a = pwmio.PWMOut(board.GP10, frequency=10000) m2b = pwmio.PWMOut(board.GP11, frequency=10000) # --- Setup HC-SR04 --- sonar = adafruit_hcsr04.HCSR04(trigger_pin=board.GP3, echo_pin=board.GP2) SPEED = 0.6 # ความเร็ว 60% def set_motor(a, b, s): if s >= 0: a.duty_cycle = int(s * 65535); b.duty_cycle = 0 else: a.duty_cycle = 0; b.duty_cycle = int(abs(s) * 65535) def forward(): set_motor(m1a, m1b, SPEED); set_motor(m2a, m2b, SPEED) def backward(): set_motor(m1a, m1b, -SPEED); set_motor(m2a, m2b, -SPEED) def turn_right(): set_motor(m1a, m1b, -SPEED); set_motor(m2a, m2b, SPEED) def turn_left(): set_motor(m1a, m1b, SPEED); set_motor(m2a, m2b, -SPEED) def stop(): set_motor(m1a, m1b, 0); set_motor(m2a, m2b, 0)

📚 Main Loop — หลบสิ่งกีดขวาง

while True: try: d = sonar.distance print(d, "cm") if d < 20: # เจอสิ่งกีดขวาง — หยุดก่อน stop() time.sleep(0.2) # ถอย 0.5 วินาที backward() time.sleep(0.5) # เลี้ยวขวา 0.5 วินาที turn_right() time.sleep(0.5) else: # ไม่มีอะไรขวาง — เดินต่อ forward() except RuntimeError: forward() # ถ้าเซ็นเซอร์เพี้ยน — เดินต่อ time.sleep(0.05)
💡
คิดเชิงวิศวกร: ทำไมต้องถอยก่อนเลี้ยว? เพราะถ้าใกล้กำแพงมาก ๆ การเลี้ยวอาจทำให้ชนข้างได้ → ถอยก่อน เพื่อให้มีพื้นที่หมุนตัว

🎯 ตัวแปรปรับแต่ง

🎨 เพิ่ม LED แสดงสถานะ

import board, neopixel, pwmio, adafruit_hcsr04, time pixels = neopixel.NeoPixel(board.GP18, 2) # ... motor setup เหมือนเดิม ... while True: try: d = sonar.distance if d < 20: pixels.fill((255, 0, 0)) # แดง = หยุด stop(); time.sleep(0.2) pixels.fill((255, 100, 0)) # ส้ม = ถอย backward(); time.sleep(0.5) pixels.fill((0, 0, 255)) # น้ำเงิน = เลี้ยว turn_right(); time.sleep(0.5) else: pixels.fill((0, 255, 0)) # เขียว = ปลอดภัย เดินหน้า forward() except RuntimeError: pass time.sleep(0.05)
🌟
เคล็ดลับ Pro: ถ้าเลี้ยวขวาแล้วเจอกำแพงอีก → ลองเลี้ยวซ้ายแทน! เพิ่มตัวแปรนับ "เลี้ยวขวากี่ครั้งแล้ว" — ถ้า ≥ 3 ครั้ง สลับเป็นเลี้ยวซ้าย
